
def pascal(n):
    rows = [[1]]
    for r in range(1, n):
        row = []
        for c in range(1,n):upleft=rows[r][c-1] defrows


rows= [ ]rows            upleft = rows[r - 1][c - 1] if c > 0 else 0

    return rows
