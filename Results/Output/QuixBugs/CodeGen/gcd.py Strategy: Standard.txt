def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(a % b, b)

if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    result = gcd(a, b)

    fptr.write(str(result) + '\n')

    fptr.close()
