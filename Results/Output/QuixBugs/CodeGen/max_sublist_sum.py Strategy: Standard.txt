
def max_sublist_sum(arr):
    max_ending_here = 0
    max_so_far = 0

    for x in arr:
        max_ending_here = max_ending_here + x
        if max_ending_here < 0:
            max_ending_here = 0

        max_so_far = max(max_so_far, max_ending_here)

    return max_so_far


print(max_sublist_sum([-2, 1, -3, 4, -1, 2, 1, -5, 4]))
