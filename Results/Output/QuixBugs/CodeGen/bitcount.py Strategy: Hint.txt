
def bitcount(n):
    count = 0
    while n:
        #buggy line:        n ^= n - 1
        n = n & (n - 1)
        count += 1
    return count

def test_bitcount():
    assert bitcount(0) == 0
    assert bitcount(1) == 1
    assert bitcount(2) == 1

    assert bitcount(5) == 2
    assert bitcount(6) == 3
    assert bitcount(7) == 3
    assert bitcount(8) == 3
