
def pascal(n):
    rows = [[1]]
    for r in range(1, n):
        row = []
        for c in range(r + 1):
            if c == 0 or c == r:
                row.append(1)
            else:
                row.append(rows[r - 1][c - 1] + rows[r - 1][c])
        rows.append(row)
    return rows


if __name__ == '__main__':
    print(pascal(5))
